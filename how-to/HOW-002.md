## Ziel

Einen Telegram-Bot aufbauen, der Fragen per RAG aus einer lokalen Markdown-Wissensbasis beantwortet, über Temporal orchestriert wird und sich bei Änderungen automatisch neu indexiert.

## Voraussetzungen

- Python 3.11+
- Docker (für Temporal)
- ngrok (für lokale Entwicklung)
- Telegram Bot Token (via @BotFather)
- Anthropic API Key (api.anthropic.com)
- Git-Repository der Wissensbasis auf dem Server verfügbar

## 2. ChromaDB-Indexierung (indexer.py)

Chunking-Strategie nach ADR-004:
- `requirements/`, `decisions/`: Datei als ein Chunk
- `how-to/`, `notes/`: Rekursiver Zeichen-Split (400 Tokens, 80 Overlap)
- `principles/principles.md`: Header-Split an `##`

```python
import os
import subprocess
from pathlib import Path
from dotenv import load_dotenv
import chromadb
from chromadb.utils.embedding_functions import SentenceTransformerEmbeddingFunction
from watchdog.observers import Observer
from watchdog.events import FileSystemEventHandler

load_dotenv()
KB_PATH = os.getenv("KB_PATH", "./knowledge")
CHROMA_PATH = os.getenv("CHROMA_PATH", "./chroma")

embedding_fn = SentenceTransformerEmbeddingFunction(
    model_name="intfloat/multilingual-e5-base"
)
client = chromadb.PersistentClient(path=CHROMA_PATH)
collection = client.get_or_create_collection(
    name="knowledge",
    embedding_function=embedding_fn,
)

SINGLE_CHUNK_DIRS = {"requirements", "decisions"}
HEADER_SPLIT_FILES = {"principles.md"}
CHUNK_SIZE = 400
CHUNK_OVERLAP = 80


def _split_recursive(text: str) -> list[str]:
    """Rekursiver Zeichen-Split an Absatz-, Satz- und Zeichengrenzen."""
    separators = ["\n\n", "\n", ". ", " ", ""]
    for sep in separators:
        parts = text.split(sep) if sep else list(text)
        chunks, current = [], ""
        for part in parts:
            candidate = current + (sep if current else "") + part
            if len(candidate) <= CHUNK_SIZE:
                current = candidate
            else:
                if current:
                    chunks.append(current)
                current = part
        if current:
            chunks.append(current)
        if all(len(c) <= CHUNK_SIZE for c in chunks):
            return _add_overlap(chunks, sep)
    return [text]


def _add_overlap(chunks: list[str], sep: str) -> list[str]:
    if len(chunks) <= 1:
        return chunks
    result = [chunks[0]]
    for i in range(1, len(chunks)):
        tail = result[-1][-CHUNK_OVERLAP:]
        result.append(tail + sep + chunks[i])
    return result


def _split_by_header(text: str) -> list[str]:
    sections, current = [], []
    for line in text.splitlines(keepends=True):
        if line.startswith("## ") and current:
            sections.append("".join(current).strip())
            current = [line]
        else:
            current.append(line)
    if current:
        sections.append("".join(current).strip())
    return [s for s in sections if s]


def _chunks_for_file(path: str) -> list[tuple[str, str]]:
    """Gibt (chunk_id, text)-Paare für eine Datei zurück."""
    rel = Path(path).relative_to(KB_PATH)
    text = Path(path).read_text(encoding="utf-8")
    top_dir = rel.parts[0] if len(rel.parts) > 1 else ""
    filename = rel.name

    if top_dir in SINGLE_CHUNK_DIRS:
        return [(str(rel), text)]
    if filename in HEADER_SPLIT_FILES:
        return [(f"{rel}#{i}", chunk) for i, chunk in enumerate(_split_by_header(text))]
    chunks = _split_recursive(text)
    return [(f"{rel}#{i}", chunk) for i, chunk in enumerate(chunks)]


def index_file(path: str):
    pairs = _chunks_for_file(path)
    ids = [p[0] for p in pairs]
    texts = [p[1] for p in pairs]
    # Alte Chunks dieser Datei entfernen
    rel = str(Path(path).relative_to(KB_PATH))
    existing = collection.get(where={"source": rel})
    if existing["ids"]:
        collection.delete(ids=existing["ids"])
    collection.upsert(
        ids=ids,
        documents=texts,
        metadatas=[{"source": rel}] * len(ids),
    )
    print(f"Indexed: {path} ({len(ids)} chunks)")


def remove_file(path: str):
    rel = str(Path(path).relative_to(KB_PATH))
    existing = collection.get(where={"source": rel})
    if existing["ids"]:
        collection.delete(ids=existing["ids"])
    print(f"Removed from index: {path}")


def reindex_all():
    for root, _, files in os.walk(KB_PATH):
        for file in files:
            if file.endswith(".md"):
                index_file(os.path.join(root, file))


class KnowledgeBaseHandler(FileSystemEventHandler):
    def on_created(self, event):
        if not event.is_directory and event.src_path.endswith(".md"):
            index_file(event.src_path)

    def on_modified(self, event):
        if not event.is_directory and event.src_path.endswith(".md"):
            index_file(event.src_path)

    def on_deleted(self, event):
        if not event.is_directory and event.src_path.endswith(".md"):
            remove_file(event.src_path)


def start_watcher() -> Observer:
    observer = Observer()
    observer.schedule(KnowledgeBaseHandler(), path=KB_PATH, recursive=True)
    observer.start()
    return observer


def pull_and_reindex():
    result = subprocess.run(
        ["git", "pull"], cwd=KB_PATH, capture_output=True, text=True
    )
    if result.returncode == 0:
        reindex_all()
    else:
        print(f"git pull failed: {result.stderr}")
```

## 3. RAG-Abfrage (rag.py)

```python
import os
from dotenv import load_dotenv
import chromadb
from chromadb.utils.embedding_functions import SentenceTransformerEmbeddingFunction
import litellm

load_dotenv()
TOP_K = int(os.getenv("TOP_K", "5"))
MAX_HISTORY = int(os.getenv("MAX_HISTORY", "10"))
MODEL_NAME = os.getenv("MODEL_NAME", "claude-haiku-4-5")
CHROMA_PATH = os.getenv("CHROMA_PATH", "./chroma")

embedding_fn = SentenceTransformerEmbeddingFunction(
    model_name="intfloat/multilingual-e5-base"
)
client = chromadb.PersistentClient(path=CHROMA_PATH)
collection = client.get_or_create_collection(
    name="knowledge",
    embedding_function=embedding_fn,
)

# Sitzungsspeicher: {chat_id: [{"role": ..., "content": ...}]} (ADR-011)
_history: dict[int, list[dict]] = {}

SYSTEM_PROMPT_PATH = os.path.join(
    os.path.dirname(__file__), "..", "config", "system-prompt.md"
)
with open(SYSTEM_PROMPT_PATH, encoding="utf-8") as f:
    SYSTEM_PROMPT = f.read()


def _get_history(chat_id: int) -> list[dict]:
    return _history.get(chat_id, [])


def _append_history(chat_id: int, role: str, content: str):
    history = _history.setdefault(chat_id, [])
    history.append({"role": role, "content": content})
    # Sliding window: nur die letzten MAX_HISTORY Nachrichten behalten
    if len(history) > MAX_HISTORY:
        _history[chat_id] = history[-MAX_HISTORY:]


def query(chat_id: int, user_message: str) -> str:
    # Retrieval
    results = collection.query(query_texts=[user_message], n_results=TOP_K)
    chunks = results["documents"][0] if results["documents"] else []

    if chunks:
        context = "\n\n---\n\n".join(chunks)
        context_note = ""
    else:
        # Fallback A: kein relevanter Chunk (ADR-010)
        context = ""
        context_note = (
            "\n\nHinweis: Kein passender Eintrag in der Wissensbasis gefunden. "
            "Die folgende Antwort basiert auf allgemeinem Wissen."
        )

    # Prompt zusammensetzen
    if context:
        user_content = (
            f"Wissensbasis-Kontext:\n{context}\n\nFrage: {user_message}"
        )
    else:
        user_content = user_message

    messages = [{"role": "system", "content": SYSTEM_PROMPT}]
    messages += _get_history(chat_id)
    messages.append({"role": "user", "content": user_content})

    # LLM-Call (Fallback B: Retry via Temporal, hier Exception werfen)
    response = litellm.completion(
        model=MODEL_NAME,
        messages=messages,
        api_key=os.getenv("LLM_API_KEY"),
    )
    answer = response.choices[0].message.content + context_note

    # Verlauf aktualisieren (mit der ursprünglichen Nutzerfrage, ohne Kontext)
    _append_history(chat_id, "user", user_message)
    _append_history(chat_id, "assistant", answer)

    return answer
```

## 4. Temporal Workflow (worker.py)

```python
import asyncio
from temporalio import activity, workflow
from temporalio.client import Client
from temporalio.worker import Worker
import requests
import os
from dotenv import load_dotenv
from rag import query

load_dotenv()
TELEGRAM_BOT_TOKEN = os.getenv("TELEGRAM_BOT_TOKEN")
TELEGRAM_API_URL = f"https://api.telegram.org/bot{TELEGRAM_BOT_TOKEN}/sendMessage"


@activity.defn
async def process_message_activity(chat_id: int, message: str) -> str:
    return query(chat_id, message)


@activity.defn
async def send_telegram_activity(chat_id: int, text: str):
    requests.post(TELEGRAM_API_URL, json={"chat_id": chat_id, "text": text})


@workflow.defn
class MessageWorkflow:
    @workflow.run
    async def run(self, chat_id: int, message: str):
        try:
            answer = await workflow.execute_activity(
                process_message_activity,
                args=[chat_id, message],
                start_to_close_timeout=__import__("datetime").timedelta(seconds=30),
                retry_policy=__import__("temporalio.common", fromlist=["RetryPolicy"]).RetryPolicy(
                    maximum_attempts=3,
                ),
            )
        except Exception:
            # Fallback B: LLM nach Retries nicht erreichbar (ADR-010)
            answer = "Der Dienst ist gerade nicht erreichbar. Bitte versuche es später erneut."

        await workflow.execute_activity(
            send_telegram_activity,
            args=[chat_id, answer],
            start_to_close_timeout=__import__("datetime").timedelta(seconds=10),
        )


async def main():
    client = await Client.connect("localhost:7233")
    worker = Worker(
        client,
        task_queue="telegram-bot",
        workflows=[MessageWorkflow],
        activities=[process_message_activity, send_telegram_activity],
    )
    await worker.run()


if __name__ == "__main__":
    asyncio.run(main())
```

## 5. Flask-Server (bot.py)

```python
import asyncio
import hmac
import hashlib
import os
import chromadb
from flask import Flask, request
from temporalio.client import Client
from dotenv import load_dotenv
from indexer import start_watcher, pull_and_reindex
from worker import MessageWorkflow

load_dotenv()
TELEGRAM_BOT_TOKEN = os.getenv("TELEGRAM_BOT_TOKEN")
GITHUB_WEBHOOK_SECRET = os.getenv("GITHUB_WEBHOOK_SECRET", "")

app = Flask(__name__)
watcher = start_watcher()


async def _start_workflow(chat_id: int, message: str):
    client = await Client.connect("localhost:7233")
    await client.start_workflow(
        MessageWorkflow.run,
        args=[chat_id, message],
        id=f"msg-{chat_id}-{__import__('time').time_ns()}",
        task_queue="telegram-bot",
    )


@app.route("/webhook", methods=["POST"])
def webhook():
    data = request.json
    if not data or "message" not in data:
        return "ok"
    chat_id = data["message"]["chat"]["id"]
    message = data["message"].get("text", "")
    if message:
        asyncio.run(_start_workflow(chat_id, message))
    return "ok"


@app.route("/github-webhook", methods=["POST"])
def github_webhook():
    # Fallback C: ChromaDB-Fehler bei Neuindexierung führt zu Fehlermeldung im Log
    if GITHUB_WEBHOOK_SECRET:
        signature = request.headers.get("X-Hub-Signature-256", "")
        expected = "sha256=" + hmac.new(
            GITHUB_WEBHOOK_SECRET.encode(), request.data, hashlib.sha256
        ).hexdigest()
        if not hmac.compare_digest(signature, expected):
            return "Unauthorized", 401
    pull_and_reindex()
    return "ok"


if __name__ == "__main__":
    app.run(port=5000)
```

```bash
docker run --rm -p 7233:7233 temporalio/auto-setup:latest
```

```bash
cd temporal
python -c "from indexer import reindex_all; reindex_all()"
```

```bash
python temporal/worker.py
```

### Bot starten
```bash
python temporal/bot.py
```

### ngrok starten (lokale Entwicklung)
```bash
ngrok http 5000
```

### Telegram-Webhook registrieren
```bash
curl -X POST "https://api.telegram.org/bot<TOKEN>/setWebhook" \
  -d "url=https://<ngrok-url>/webhook"
```

### GitHub Webhook einrichten
Im GitHub-Repository unter Settings → Webhooks:
- Payload URL: `https://<server-url>/github-webhook`
- Content type: `application/json`
- Secret: Wert aus `GITHUB_WEBHOOK_SECRET`
- Event: "Just the push event"

## 7. Fallback-Verhalten (ADR-010)

| Szenario | Verhalten |
|---|---|
| Kein Chunk gefunden | LLM antwortet + Hinweis „Kein Eintrag gefunden" am Ende |
| LLM-API nicht erreichbar | Temporal wiederholt bis zu 3×, danach Fehlermeldung an Nutzer |
| ChromaDB nicht erreichbar | Exception in `rag.py` propagiert zu Temporal → Fehlermeldung |
